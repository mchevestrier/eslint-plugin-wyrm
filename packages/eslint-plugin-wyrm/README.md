## Installation

```shell
npm install --save-dev eslint eslint-plugin-wyrm
```

## Usage

```ts
import { defineConfig } from 'eslint/config';
import wyrm from 'eslint-plugin-wyrm';

export default defineConfig([
  // ...

  wyrm.configs.recommended,
]);
```
